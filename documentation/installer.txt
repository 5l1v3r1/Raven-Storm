echo "[i] Installation"
if [[ "$OSTYPE" == "linux-gnu" ]]; then
        sudo apt-get -y upgrade
        git --help > /dev/null || sudo apt-get -y install git || echo "[!] Could not install git"
        python2 --help > /dev/null || sudo apt-get -y install python2 || echo "[!] Could not install python2"
        ping -h > /dev/null || sudo apt-get -y install iputils-ping || echo "[!] Could not install ping"
        redis --help > /dev/null || sudo apt-get -y install redis || echo "[!] Could not install redis"
        nmap --help > /dev/null || sudo apt-get -y install nmap || echo "[!] Could not install nmap"
        git clone https://github.com/Taguar258/Raven-Storm/
        cd Raven-Storm
        python2 -m pip install -r requirements.txt
        echo "[i] Done"
        echo "[i] Run: >python2 rs.pyo< to start"
elif [[ "$OSTYPE" == "darwin"* ]]; then
        brew --help > /dev/null || /usr/bin/ruby -e "$(curl -fsSL https://raw.githubusercontent.com/Homebrew/install/master/install)"
        git --help > /dev/null || brew install git || echo "[!] Could not install git"
        python2 --help > /dev/null || brew install python2 || echo "[!] Could not install python2"
        ping -h > /dev/null || brew install iputils-ping || echo "[!] Could not install ping"
        redis --help > /dev/null || brew install redis || echo "[!] Could not install redis"
        nmap --help > /dev/null || brew install nmap || echo "[!] Could not install nmap"
        git clone https://github.com/Taguar258/Raven-Storm/
        cd Raven-Storm
        python2 -m pip install -r requirements.txt
        echo "[i] Done"
        echo "[i] Run: >python2 rs.pyo< to start"
elif [[ "$OSTYPE" == "cygwin" ]]; then
        # POSIX compatibility layer and Linux environment emulation for Windows
        echo "[i] Please install manually"
elif [[ "$OSTYPE" == "msys" ]]; then
        # Lightweight shell and GNU utilities compiled for Windows (part of MinGW)
        echo "[i] Please install manually"
elif [[ "$OSTYPE" == "win32" ]]; then
        # I'm not sure this can happen.
        echo "[i] Please install manually"
elif [ -f "/etc/arch-release" ]; then
        git --help > /dev/null || sudo pacman -S git || echo "[!] Could not install git"
        python2 --help > /dev/null || sudo pacman -S python2 || echo "[!] Could not install python2"
        ping -h > /dev/null || sudo pacman -S iputils-ping || echo "[!] Could not install ping"
        redis --help > /dev/null || sudo pacman -S redis || echo "[!] Could not install redis"
        nmap --help > /dev/null || sudo pacman -S nmap || echo "[!] Could not install nmap"
        git clone https://github.com/Taguar258/Raven-Storm/
        cd Raven-Storm
        python2 -m pip install -r requirements.txt
        echo "[i] Done"
        echo "[i] Run: >python2 rs.pyo< to start"
else
        echo "[i] Please install manually"
fi
